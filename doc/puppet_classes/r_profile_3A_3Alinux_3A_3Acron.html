Overview

R_profile::Linux::Cron

Restrict permissions on the cron scheduler

  • Remove cron.deny so that only allowed users can run cron
  • Strict permissions on at.allow
  • Allow root to use cron
  • All files under /var/spool/cron owned by root with 0600 permissions (crontab -e will relax these)

# File 'manifests/linux/cron.pp', line 12

class r_profile::linux::cron {

  file { "/etc/cron.deny":
    ensure => absent,
  }

  file { "/etc/crontab":
    ensure => file,
    owner  => "root",
    group  => "root",
    mode   => "0400",
  }

  file { "/etc/cron.allow":
    ensure => file,
    owner  => "root",
    group  => "root",
    mode   => "0400",
  }

  chmod_r { "/var/spool/cron":
    want_mode => "0600",
  }

  chown_r { "/var/spool/cron":
    want_user  => "root",
    want_group => "root",
  }

  file_line { "/etc/cron.allow root":
    ensure => present,
    line   => "root",
    path   => "/etc/cron.allow"
  }
}
