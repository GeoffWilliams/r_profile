class r_profile::database::mysql_server(
    Hash[String, Any]               $settings = {},
    Hash[String, Hash[String, Any]] $dbs      = {},
) {

  class { 'mysql::server':
    * => $settings,
  }

  $dbs.each |$key, $opts| {
    mysql::db { $key:
      * => $opts,
    }
  }

}
