class r_profile::linux::module_disable(
    Array[String] $modules        = [],
    String        $disable_file = "/etc/modprobe.d/disabled.conf",
) {

  file { $disable_file:
    ensure => file,
    owner  => "root",
    group  => "root",
    mode   => "0644",
  }

  $modules.each |$module| {

    file_line { "${disable_file} disable_${module}":
      ensure => present,
      path   => $disable_file,
      match  => "^install ${module}",
      line   => "install ${module} /bin/true",
    }
  }

}
